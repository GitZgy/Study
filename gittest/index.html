1创建git 用户名和密码
   $ git config --global user.name "GitZgy"
$ git config --global user.email "1291037223@qq.com"
1创建git 仓库
   $ mkdir D:/git
$ cd D:/git
$ git init
1创建git 文件
   $ echo "hello word" > index.html
1增加git监视
   $ git add index.html
1查看git监视状态
   $ git status
1查看git文件改动
   $ git diff index.html
1提交改动的git文件
   $ git commit -m "add modiefy"
1查看日志
   $ git log
1版本回溯
   $ git reset --hard HEAD^
1版本回到未来
   $ git reflog
$ git reset --hard commit_id
